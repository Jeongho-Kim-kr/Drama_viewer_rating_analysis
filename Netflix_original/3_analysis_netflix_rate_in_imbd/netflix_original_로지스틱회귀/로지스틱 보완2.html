from statsmodels.stats.outliers_influence import variance_inflation_factor

def X_filter_multicollinearity(X, thresh=10.0):
    variables = list(range(X.shape[1]))
    dropped = True
    
    while dropped:
        dropped = False
        vif = [variance_inflation_factor(X.iloc[:, variables].values, i) for i in range(X.iloc[:, variables].shape[1])]
        
        for j in variables:
            print(X.columns[j], ':', vif[variables.index(j)])
            
        maxloc = vif.index(max(vif))

        if max(vif) > thresh:
            print('==> [Dropped variable] : ' + X.iloc[:, variables].columns[maxloc])
            del variables[maxloc]    

            if len(variables) > 1:
                dropped = True
                
        print('-'*50)
        print(' ')

    print('[Remaining variables] :')
    print(X.columns[variables])

    return variables

Precision, Recall, F1-Score¶

Precision = TP / ( FP + TP )
예측을 positive로 한 대상 중 예측과 실제 값이 positive로 일치하는 비율

Recall = TP / ( TP + FN )
실제값이 positive인 대상 중에 예측과 실제 값이 positive로 일치하는 데이터의 비율

F1-Score = Precision * Recall / ( Precision + Recall )
